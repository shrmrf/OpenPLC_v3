180 template <class T>
181 IINField DatabaseBuffers::GenericSelect(
182  Range range,
183  openpal::ArrayView<Cell<T>, uint16_t> view,
184  bool useDefault,
185  typename T::static_variation_t variation)
186 {
187  if (range.IsValid())
188  {
189  auto allowed = range.Intersection(RangeOf(view.Size()));
190 
191  if (allowed.IsValid())
192  {
193  // return code depends on if the range was truncated to match the database
194  IINField ret = allowed.Equals(range) ? IINField() : IINBit::PARAM_ERROR;
195 
196  for (uint16_t i = allowed.start; i <= allowed.stop; ++i)
197  {
198  if (view[i].selection.selected)
199  {
200  ret |= IINBit::PARAM_ERROR;
201  }
202  else
203  {
204  view[i].selection.selected = true;
205  view[i].selection.value = view[i].value;
206  auto var = useDefault ? view[i].config.svariation : variation;
207  view[i].selection.variation = CheckForPromotion<T>(view[i].selection.value, var);
208  }
209  }
210 
211  ranges.Merge<T>(allowed);
212 
213  return ret;
214  }
215  else
216  {
217  return IINField(IINBit::PARAM_ERROR);
218  }
219  }
220  else
221  {
222  return IINField();
223  }
224 }

226 template <class T>
227 bool DatabaseBuffers::LoadType(HeaderWriter& writer)
228 {
229  auto range = ranges.Get<T>();
230  if (range.IsValid())
231  {
232  auto view = buffers.GetArrayView<T>();
233 
234  bool spaceRemaining = true;
235 
236  // ... load values, manipulate the range
237  while (spaceRemaining && range.IsValid())
238  {
239  if (view[range.start].selection.selected)
240  {
242  auto writeFun = GetStaticWriter(view[range.start].selection.variation);
243 
244  // start writing a header, the invoked function will advance the range appropriately
245  spaceRemaining = writeFun(view, writer, range);
246  }
247  else
248  {
249  // just skip over values that are not selected
250  range.Advance();
251  }
252  }
253 
254  ranges.Set<T>(range);
255 
256  return spaceRemaining;
257  }
258  else
259  {
260  // no data to load
261  return true;
262  }
263 }
