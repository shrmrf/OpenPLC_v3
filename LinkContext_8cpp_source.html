64 bool LinkContext::OnLowerLayerUp()
65 {

132 bool LinkContext::OnTransmitResult(bool success)
133 {
134  if (this->txMode == LinkTransmitMode::Idle)
135  {
136  SIMPLE_LOG_BLOCK(this->logger, flags::ERR, "Unknown transmission callback");
137  return false;
138  }
139 
140  auto isPrimary = (this->txMode == LinkTransmitMode::Primary);
141  this->txMode = LinkTransmitMode::Idle;
142 
143  // before we dispatch the transmit result, give any pending transmissions access first
144  this->TryPendingTx(this->pendingSecTx, false);
145  this->TryPendingTx(this->pendingPriTx, true);
146 
147  // now dispatch the completion event to the correct state handler
148  if (isPrimary)
149  {
150  this->pPriState = &this->pPriState->OnTransmitResult(*this, success);
151  }
152  else
153  {
154  this->pSecState = &this->pSecState->OnTransmitResult(*this, success);
155  }
156 
157  return true;
158 }

281  auto elapsed = now.milliseconds - this->lastMessageTimestamp.milliseconds;

340 bool LinkContext::OnFrame(const LinkHeaderFields& header, const openpal::RSlice& userdata)
341 {
342  if (!isOnline)
343  {
344  SIMPLE_LOG_BLOCK(logger, flags::ERR, "Layer is not online");
345  return false;
346  }
347 
348  if (!this->Validate(header.isFromMaster, header.src, header.dest))
349  {
350  return false;
351  }
352 
353  // reset the keep-alive timestamp
354  this->lastMessageTimestamp = this->executor->GetTime();
355 
356  switch (header.func)
357  {
358  case(LinkFunction::SEC_ACK) :
359  pPriState = &pPriState->OnAck(*this, header.fcvdfc);
360  return true;
361  case(LinkFunction::SEC_NACK) :
362  pPriState = &pPriState->OnNack(*this, header.fcvdfc);
363  return true;
364  case(LinkFunction::SEC_LINK_STATUS) :
365  pPriState = &pPriState->OnLinkStatus(*this, header.fcvdfc);
366  return true;
367  case(LinkFunction::SEC_NOT_SUPPORTED) :
368  pPriState = &pPriState->OnNotSupported(*this, header.fcvdfc);
369  return true;
370  case(LinkFunction::PRI_TEST_LINK_STATES) :
371  pSecState = &pSecState->OnTestLinkStatus(*this, header.fcb);
372  return true;
373  case(LinkFunction::PRI_RESET_LINK_STATES) :
374  pSecState = &pSecState->OnResetLinkStates(*this);
375  return true;
376  case(LinkFunction::PRI_REQUEST_LINK_STATUS) :
377  pSecState = &pSecState->OnRequestLinkStatus(*this);
378  return true;
379  case(LinkFunction::PRI_CONFIRMED_USER_DATA) :
380  pSecState = &pSecState->OnConfirmedUserData(*this, header.fcb, userdata);
381  return true;
382  case(LinkFunction::PRI_UNCONFIRMED_USER_DATA) :
383  this->PushDataUp(userdata);
384  return true;
385  default:
386  return false;
387  }
388 }

390 bool LinkContext::Validate(bool isMaster, uint16_t src, uint16_t dest)
391 {
392  if (isMaster == config.IsMaster)
393  {
394  ++statistics.numBadMasterBit;
395  SIMPLE_LOG_BLOCK(logger, flags::WARN, (isMaster ? "Master frame received for master" : "Outstation frame received for outstation"));
396  return false;
397  }
398 
399  if (dest != config.LocalAddr)
400  {
401  ++statistics.numUnknownDestination;
402  SIMPLE_LOG_BLOCK(logger, flags::WARN, "Frame for unknown destintation");
403  return false;
404  }
405 
406  if (src != config.RemoteAddr)
407  {
408  ++statistics.numUnknownSource;
409  SIMPLE_LOG_BLOCK(logger, flags::WARN, "Frame from unknwon source");
410  return false;
411  }
412 
413  return true;
414 }
