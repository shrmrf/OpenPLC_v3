132 static uint16_t crc16(uint8_t *buffer, uint16_t buffer_length)
133 {
134  uint8_t crc_hi = 0xFF; /* high CRC byte initialized */
135  uint8_t crc_lo = 0xFF; /* low CRC byte initialized */
136  unsigned int i; /* will index into CRC lookup */
137 
138  /* pass through message buffer */
139  while (buffer_length--) {
140  i = crc_hi ^ *buffer++; /* calculate the CRC */
141  crc_hi = crc_lo ^ table_crc_hi[i];
142  crc_lo = table_crc_lo[i];
143  }
144 
145  return (crc_hi << 8 | crc_lo);
146 }

305 {
306  int rc;
307  modbus_rtu_t *ctx_rtu = ctx->backend_data;
308 
309  if (ctx_rtu->confirmation_to_ignore) {
310  _modbus_receive_msg(ctx, req, MSG_CONFIRMATION);
311  /* Ignore errors and reset the flag */
312  ctx_rtu->confirmation_to_ignore = FALSE;
313  rc = 0;
314  if (ctx->debug) {
315  printf("Confirmation to ignore\n");
316  }
317  } else {
318  rc = _modbus_receive_msg(ctx, req, MSG_INDICATION);
319  if (rc == 0) {
320  /* The next expected message is a confirmation to ignore */
321  ctx_rtu->confirmation_to_ignore = TRUE;
322  }
323  }
324  return rc;
325 }

356 /* The check_crc16 function shall return 0 is the message is ignored and the
357  message length if the CRC is valid. Otherwise it shall return -1 and set
358  errno to EMBADCRC. */
359 static int _modbus_rtu_check_integrity(modbus_t *ctx, uint8_t *msg,
360  const int msg_length)
361 {
362  uint16_t crc_calculated;
363  uint16_t crc_received;
364  int slave = msg[0];
365 
366  /* Filter on the Modbus unit identifier (slave) in RTU mode to avoid useless
367  * CRC computing. */
368  if (slave != ctx->slave && slave != MODBUS_BROADCAST_ADDRESS) {
369  if (ctx->debug) {
370  printf("Request for slave %d ignored (not %d)\n", slave, ctx->slave);
371  }
372  /* Following call to check_confirmation handles this error */
373  return 0;
374  }
375 
376  crc_calculated = crc16(msg, msg_length - 2);
377  crc_received = (msg[msg_length - 2] << 8) | msg[msg_length - 1];
378 
379  /* Check CRC of msg */
380  if (crc_calculated == crc_received) {
381  return msg_length;
382  } else {
383  if (ctx->debug) {
384  fprintf(stderr, "ERROR CRC received 0x%0X != CRC calculated 0x%0X\n",
385  crc_received, crc_calculated);
386  }
387 
388  if (ctx->error_recovery & MODBUS_ERROR_RECOVERY_PROTOCOL) {
389  _modbus_rtu_flush(ctx);
390  }
391  errno = EMBBADCRC;
392  return -1;
393  }
394 }

848  /* C_CC Control characters
849  VMIN Minimum number of characters to read
850  VTIME Time to wait for data (tenths of seconds)
851 
852  UNIX serial interface drivers provide the ability to
853  specify character and packet timeouts. Two elements of the
854  c_cc array are used for timeouts: VMIN and VTIME. Timeouts
855  are ignored in canonical input mode or when the NDELAY
856  option is set on the file via open or fcntl.
857 
858  VMIN specifies the minimum number of characters to read. If
859  it is set to 0, then the VTIME value specifies the time to
860  wait for every character read. Note that this does not mean
861  that a read call for N bytes will wait for N characters to
862  come in. Rather, the timeout will apply to the first
863  character and the read call will return the number of
864  characters immediately available (up to the number you
865  request).
866 
867  If VMIN is non-zero, VTIME specifies the time to wait for
868  the first character read. If a character is read within the
869  time given, any read will block (wait) until all VMIN
870  characters are read. That is, once the first character is
871  read, the serial interface driver expects to receive an
872  entire packet of characters (VMIN bytes total). If no
873  character is read within the time allowed, then the call to
874  read returns 0. This method allows you to tell the serial
875  driver you need exactly N bytes and any read call will
876  return 0 or N bytes. However, the timeout only applies to
877  the first character read, so if for some reason the driver
878  misses one character inside the N byte packet then the read
879  call could block forever waiting for additional input
880  characters.
881 
882  VTIME specifies the amount of time to wait for incoming
883  characters in tenths of seconds. If VTIME is set to 0 (the
884  default), reads will block (wait) indefinitely unless the
885  NDELAY option is set on the port with open or fcntl.
886  */
887  /* Unused because we use open with the NDELAY option */
888  tios.c_cc[VMIN] = 0;
889  tios.c_cc[VTIME] = 0;
890 
891  if (tcsetattr(ctx->s, TCSANOW, &tios) < 0) {
892  close(ctx->s);
893  ctx->s = -1;
894  return -1;
895  }
896 #endif
897 
898  return 0;
899 }

1210 modbus_t* modbus_new_rtu(const char *device,
1211  int baud, char parity, int data_bit,
1212  int stop_bit)
1213 {

1215  modbus_rtu_t *ctx_rtu;
1216 
1217  /* Check device argument */
1218  if (device == NULL || *device == 0) {
1219  fprintf(stderr, "The device string is empty\n");
1220  errno = EINVAL;
1221  return NULL;
1222  }
1223 
1224  /* Check baud argument */
1225  if (baud == 0) {
1226  fprintf(stderr, "The baud rate value must not be zero\n");
1227  errno = EINVAL;
1228  return NULL;
1229  }
1230 
1231  ctx = (modbus_t *)malloc(sizeof(modbus_t));
1232  _modbus_init_common(ctx);
1233  ctx->backend = &_modbus_rtu_backend;
1234  ctx->backend_data = (modbus_rtu_t *)malloc(sizeof(modbus_rtu_t));
1235  ctx_rtu = (modbus_rtu_t *)ctx->backend_data;
1236  ctx_rtu->device = NULL;
1237 
1238  /* Device name and \0 */
1239  ctx_rtu->device = (char *)malloc((strlen(device) + 1) * sizeof(char));
1240  strcpy(ctx_rtu->device, device);
1241 
1242  ctx_rtu->baud = baud;
1243  if (parity == 'N' || parity == 'E' || parity == 'O') {
1244  ctx_rtu->parity = parity;
1245  } else {
1246  modbus_free(ctx);
1247  errno = EINVAL;
1248  return NULL;
