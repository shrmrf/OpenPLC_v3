197 {
198  auto rawStart = GetRawIndex<Spec>(start);
199  auto rawStop = GetRawIndex<Spec>(stop);
200 
201  auto view = buffers.buffers.GetArrayView<Spec>();
202 
203  if (view.Contains(rawStart) && view.Contains(rawStop) && (rawStart <= rawStop))
204  {
205  for (uint16_t i = rawStart; i <= rawStop; ++i)
206  {
207  auto copy = view[i].value;
208  copy.flags = flags;
209  this->UpdateAny(view[i], copy, EventMode::Detect);
210  }
211 
212  return true;
213  }
