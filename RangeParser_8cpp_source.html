163 ParseResult RangeParser::ParseRangeOfOctetData(openpal::RSlice& buffer, const HeaderRecord& record, const Range& range, openpal::Logger* pLogger, IAPDUHandler* pHandler)
164 {
165  if (record.variation > 0)
166  {
167  const auto COUNT = range.Count();
168  uint32_t size = record.variation * COUNT;
169  if (buffer.Size() < size)
170  {
171  SIMPLE_LOGGER_BLOCK(pLogger, flags::WARN, "Not enough data for specified octet objects");
172  return ParseResult::NOT_ENOUGH_DATA_FOR_OBJECTS;
173  }
174  else
175  {
176  if (pHandler)
177  {
178  auto read = [range, record](openpal::RSlice & buffer, uint32_t pos) -> Indexed<OctetString>
179  {
180  OctetString octets(buffer.Take(record.variation));
181  buffer.Advance(record.variation);
182  return WithIndex(octets, range.start + pos);
183  };
184 
185  auto collection = CreateBufferedCollection<Indexed<OctetString>>(buffer, COUNT, read);
186 
187  pHandler->OnHeader(RangeHeader(record, range), collection);
188  }
189 
190  buffer.Advance(size);
191  return ParseResult::OK;
192  }
193  }
194  else
195  {
196  SIMPLE_LOGGER_BLOCK(pLogger, flags::WARN, "Octet string variation 0 may only be used in requests");
197  return ParseResult::INVALID_OBJECT;
198  }
199 }
