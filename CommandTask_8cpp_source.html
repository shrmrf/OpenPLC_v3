103 IMasterTask::ResponseResult CommandTask::ProcessResponse(const openpal::RSlice& objects)
104 {
105  if (functionCodes.empty())
106  {
107  auto result = CommandSetOps::ProcessOperateResponse(commands, objects, &logger);
108  return (result == CommandSetOps::OperateResult::FAIL_PARSE) ? ResponseResult::ERROR_BAD_RESPONSE : ResponseResult::OK_FINAL;
109  }
110  else
111  {
112  auto result = CommandSetOps::ProcessSelectResponse(commands, objects, &logger);
113 
114  switch (result)
115  {
116  case(CommandSetOps::SelectResult::OK) :
117  return ResponseResult::OK_REPEAT; // proceed to OPERATE
118  case(CommandSetOps::SelectResult::FAIL_SELECT) :
119  return ResponseResult::OK_FINAL; // end the task, let the user examine each command point
120  default:
121  return ResponseResult::ERROR_BAD_RESPONSE;
122  }
123  }
124 }
