29 {
30  // if one of the tasks is disabled, pick the enabled one
31

33  bool rightEnabled = TaskComparison::Enabled(rhs, filter);
34 
35  if (!leftEnabled || !rightEnabled)
36  {
37  // always prefer the enabled task over the one that is disabled
38  return rightEnabled ? Result::Right : Result::Left;
39  }
40 
41  if ((lhs.Priority() > rhs.Priority()) && rhs.BlocksLowerPriority())
42  {

44  }
45  else if (rhs.Priority() > lhs.Priority() && lhs.BlocksLowerPriority())
46  {
47  return Result::Left;
48  }
