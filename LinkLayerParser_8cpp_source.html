56 void LinkLayerParser::OnRead(uint32_t numBytes, IFrameSink& sink)
57 {
58  buffer.AdvanceWrite(numBytes);
59 
60  while (ParseUntilComplete() == State::Complete)

62  ++statistics.numLinkFrameRx;

64  state = State::FindSync;

70 LinkLayerParser::State LinkLayerParser::ParseUntilComplete()
71 {
72  auto lastState = this->state;
73  // continue as long as we're making progress, i.e. a state change
74  while ((this->state = ParseOneStep()) != lastState)
75  {
76  lastState = state;
77  }
78  return state;

227 bool LinkLayerParser::ValidateHeaderParameters()
228 {
229  if(!header.ValidLength())
230  {
231  ++statistics.numBadLength;
232  FORMAT_LOG_BLOCK(logger, flags::ERR, "LENGTH out of range [5,255]: %i", header.GetLength());
233  return false;
234  }
235 
236  //Now make sure that the function code is known and that the FCV is appropriate
237  if (!this->ValidateFunctionCode())
238  {
239  return false;
240  }
241 
242  uint8_t user_data_length = header.GetLength() - LPDU_MIN_LENGTH;
243  frameSize = LinkFrame::CalcFrameSize(user_data_length);
244  LinkFunction func = header.GetFuncEnum();
245 
246  const bool has_payload = user_data_length > 0;
247  const bool should_have_payload = (func == LinkFunction::PRI_CONFIRMED_USER_DATA || func == LinkFunction::PRI_UNCONFIRMED_USER_DATA);
248 
249  // make sure that the presence/absence of user data matches the function code
250  if(should_have_payload && !has_payload)
251  {
252  ++statistics.numBadLength;
253  FORMAT_LOG_BLOCK(logger, flags::ERR, "User data with no payload. FUNCTION: %s", LinkFunctionToString(func));
254  return false;
255  }
256 
257  if (!should_have_payload && has_payload)
258  {
259  ++statistics.numBadLength;
260  FORMAT_LOG_BLOCK(logger, flags::ERR, "Unexpected LENGTH in frame: %i with FUNCTION: %s", user_data_length, LinkFunctionToString(func));
261  return false;
262  }
263 
264  // calculate the total frame size
265  frameSize = LinkFrame::CalcFrameSize(user_data_length);
266 
267  return true;
268 }

276 bool LinkLayerParser::ValidateFunctionCode()
277 {
278  //Now make sure that the function code is known and that the FCV is appropriate
279  if(header.IsPriToSec())
280  {
281  bool fcv_set = false;
282 
283  switch(header.GetFuncEnum())
284  {
285  case(LinkFunction::PRI_CONFIRMED_USER_DATA):
286  case(LinkFunction::PRI_TEST_LINK_STATES):
287  fcv_set = true;
288  break;
289  case(LinkFunction::PRI_REQUEST_LINK_STATUS):
290  case(LinkFunction::PRI_RESET_LINK_STATES):
291  case(LinkFunction::PRI_UNCONFIRMED_USER_DATA):
292  break;
293  default:
294  {
295  ++statistics.numBadFunctionCode;
296  FORMAT_LOG_BLOCK(logger, flags::WARN, "Unknown PriToSec FUNCTION: %s", LinkFunctionToString(header.GetFuncEnum()));
297  return false;
298  }
299  }
300 
301  //now check the fcv
302  if(fcv_set != header.IsFcvDfcSet())
303  {
304  ++statistics.numBadFCV;
305  FORMAT_LOG_BLOCK(logger, flags::WARN, "Bad FCV for FUNCTION: %s", LinkFunctionToString(header.GetFuncEnum()));
306  return false;
307  }
308 
309  //if fcv isn't expected to be set, fcb can be either 1 or 0, doesn't matter
310 
311  }
312  else // SecToPri - just validate the function codes and that FCB is 0
313  {
314  switch(header.GetFuncEnum())
315  {
316  case(LinkFunction::SEC_ACK):
317  case(LinkFunction::SEC_NACK):
318  case(LinkFunction::SEC_LINK_STATUS):
319  case(LinkFunction::SEC_NOT_SUPPORTED):
320  break;
321  default:
322  {
323  ++statistics.numBadFunctionCode;
324  FORMAT_LOG_BLOCK(logger, flags::ERR, "Unknown SecToPri FUNCTION: %s", LinkFunctionToString(header.GetFuncEnum()));
325  return false;
326  }
327  }
328 
329  //now check the fcb, it should always be zero
330  if(header.IsFcbSet())
331  {
332  ++statistics.numBadFCB;
333  FORMAT_LOG_BLOCK(logger, flags::ERR, "FCB set for SecToPri FUNCTION: %s", LinkFunctionToString(header.GetFuncEnum()));
334  return false;
335  }
336  }
337 
338  return true; //valid!
339 }
