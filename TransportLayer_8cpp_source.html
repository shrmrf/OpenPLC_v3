50 bool TransportLayer::BeginTransmit(const RSlice& apdu)
51 {
52  if (!isOnline)
53  {
54  SIMPLE_LOG_BLOCK(logger, flags::ERR, "Layer offline");
55  return false;
56  }
57 
58  if (apdu.IsEmpty())
59  {
60  SIMPLE_LOG_BLOCK(logger, flags::ERR, "APDU cannot be empty");
61  return false;
62  }
63 
64  if (isSending)
65  {
66  SIMPLE_LOG_BLOCK(logger, flags::ERR, "Invalid BeginTransmit call, already transmitting");
67  return false;
68  }
69 
70  if (!lower)
71  {
72  SIMPLE_LOG_BLOCK(logger, flags::ERR, "Can't send without an attached link layer");
73  return false;
74  }
75 
76  isSending = true;
77  transmitter.Configure(apdu);
78  lower->Send(transmitter);
79 
80  return true;
81 }
