47 IMasterTask::ResponseResult ClearRestartTask::ProcessResponse(const APDUResponseHeader& response, const openpal::RSlice& objects)
48 {
49  // we only care that the response to this has FIR/FIN
50  if (ValidateSingleResponse(response))
51  {
52  if (response.IIN.IsSet(IINBit::DEVICE_RESTART))
53  {
54  // we tried to clear the restart, but the device responded with the restart still set
55  SIMPLE_LOG_BLOCK(logger, flags::ERR, "Clear restart task failed to clear restart bit, permanently disabling task");
56  return ResponseResult::ERROR_BAD_RESPONSE;
57  }
58  else
59  {
60  return ResponseResult::OK_FINAL;
61  }
62  }
63  else
64  {
65  return ResponseResult::ERROR_BAD_RESPONSE;
66  }
67 }
