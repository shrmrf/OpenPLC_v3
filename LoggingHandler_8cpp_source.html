46 std::string LoggingHandler::ToUTCString(const DNPTime& dnptime)
47 {
48  time_t seconds = static_cast<time_t>(dnptime / 1000);
49  uint16_t milliseconds = static_cast<uint16_t>(dnptime % 1000);
50 
51 #ifdef WIN32
52  tm t;
53  if (gmtime_s(&t, &seconds))
54  {
55  return "BAD TIME";
56  }
57 #else
58  tm t;
59  if (!gmtime_r(&seconds, &t))
60  {
61  return "BAD TIME";
62  }
63 #endif
64 
65  std::ostringstream oss;
66  oss << (1900 + t.tm_year);
67  oss << "-" << std::setfill('0') << std::setw(2) << (1 + t.tm_mon);
68  oss << "-" << std::setfill('0') << std::setw(2) << t.tm_mday;
69  oss << " " << std::setfill('0') << std::setw(2) << t.tm_hour;
70  oss << ":" << std::setfill('0') << std::setw(2) << t.tm_min;
71  oss << ":" << std::setfill('0') << std::setw(2) << t.tm_sec;
72  oss << "." << std::setfill('0') << std::setw(3) << milliseconds;
73  return oss.str();
74 }
