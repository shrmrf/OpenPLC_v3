225 void MContext::ProcessIIN(const IINField& iin)
226 {
227  if (iin.IsSet(IINBit::DEVICE_RESTART) && !this->params.ignoreRestartIIN)
228  {
229  this->tasks.clearRestart->Demand();
230  this->tasks.assignClass->Demand();
231  this->tasks.startupIntegrity->Demand();
232  this->tasks.enableUnsol->Demand();
233  }
234 
235  if (iin.IsSet(IINBit::EVENT_BUFFER_OVERFLOW) && this->params.integrityOnEventOverflowIIN)
236  {
237  this->tasks.startupIntegrity->Demand();
238  }
239 
240  if (iin.IsSet(IINBit::NEED_TIME))
241  {
242  switch (this->params.timeSyncMode)
243  {
244  case(TimeSyncMode::SerialTimeSync):
245  this->tasks.timeSync->Demand();
246  break;
247  default:
248  break;
249  }
250  }
251 
252  if ((iin.IsSet(IINBit::CLASS1_EVENTS) && this->params.eventScanOnEventsAvailableClassMask.HasClass1()) ||
253  (iin.IsSet(IINBit::CLASS2_EVENTS) && this->params.eventScanOnEventsAvailableClassMask.HasClass2()) ||
254  (iin.IsSet(IINBit::CLASS3_EVENTS) && this->params.eventScanOnEventsAvailableClassMask.HasClass3()))
255  {
256  this->tasks.eventScan->Demand();
257  }
258 
259  this->application->OnReceiveIIN(iin);
260 }

262 void MContext::ProcessUnsolicitedResponse(const APDUResponseHeader& header, const RSlice& objects)
263 {
264  if (!header.control.UNS)
265  {
266  SIMPLE_LOG_BLOCK(logger, flags::WARN, "Ignoring unsolicited response without UNS bit set");
267  return;
268  }
269 
270  auto result = MeasurementHandler::ProcessMeasurements(objects, logger, SOEHandler.get());
271 
272  if ((result == ParseResult::OK) && header.control.CON)
273  {
274  this->QueueConfirm(APDUHeader::UnsolicitedConfirm(header.control.SEQ));
275  }
276 
277  this->ProcessIIN(header.IIN);
278 }

447 void MContext::ScheduleAdhocTask(const std::shared_ptr<IMasterTask>& task)
448 {
449  const auto NOW = this->executor->GetTime();
450 
451  task->ConfigureStartExpiration(NOW.Add(params.taskStartTimeout));
452 
453  if (this->isOnline)
454  {

469  task->OnLowerLayerClose(NOW);

481 MContext::TaskState MContext::ResumeActiveTask()
482 {
483  if (!this->pTaskLock->Acquire(*this))
484  {
485  return TaskState::TASK_READY;
486  }
487 
488  APDURequest request(this->txBuffer.GetWSlice());
489 
491  if (!this->activeTask->BuildRequest(request, this->solSeq))
492  {
493  activeTask->OnInternalError(executor->GetTime());
494  this->CompleteActiveTask();

585 MContext::TaskState MContext::OnResponse_WaitForResponse(const APDUResponseHeader& header, const RSlice& objects)
586 {
587  if (header.control.SEQ != this->solSeq)
588  {
589  FORMAT_LOG_BLOCK(this->logger, flags::WARN, "Response with bad sequence: %u", header.control.SEQ);
590  return TaskState::WAIT_FOR_RESPONSE;
591  }
592 
593  this->responseTimer.Cancel();
594 
595  this->solSeq.Increment();
596 
597  auto now = this->executor->GetTime();
598 
599  auto result = this->activeTask->OnResponse(header, objects, now);
600 
601  if (header.control.CON)
602  {
603  this->QueueConfirm(APDUHeader::SolicitedConfirm(header.control.SEQ));
604  }
605 
606  switch (result)
607  {
608  case(IMasterTask::ResponseResult::OK_CONTINUE) :
609  this->StartResponseTimer();
610  return TaskState::WAIT_FOR_RESPONSE;
611  case(IMasterTask::ResponseResult::OK_REPEAT) :
612  return StartTask_TaskReady();
613  default:
614  // task completed or failed, either way go back to idle
615  this->CompleteActiveTask();
616  return TaskState::IDLE;
617  }
618 }
