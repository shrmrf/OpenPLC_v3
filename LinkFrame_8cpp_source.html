38 void LinkFrame::ReadUserData(const uint8_t* pSrc, uint8_t* pDest, uint32_t length_)
39 {
40  uint32_t length = length_;
41  uint8_t const* pRead = pSrc;
42  uint8_t* pWrite = pDest;
43 
44  while (length > 0)
45  {
46  uint32_t max = LPDU_DATA_BLOCK_SIZE;
47  uint32_t num = (length <= max) ? length : max;
48  uint32_t num_with_crc = num + 2;

50  pRead += num_with_crc;
51  pWrite += num;
52  length -= num;
53  }

82 uint32_t LinkFrame::CalcUserDataSize(uint8_t dataLength)
83 {
84  if (dataLength > 0)
85  {
86  uint32_t mod16 = dataLength % LPDU_DATA_BLOCK_SIZE;
87  uint32_t size = (dataLength / LPDU_DATA_BLOCK_SIZE) * LPDU_DATA_PLUS_CRC_SIZE; //complete blocks
88  return (mod16 > 0) ? (size + mod16 + LPDU_CRC_SIZE) : size; //possible partial block
89  }
90  else
91  {
92  return 0;
93  }

185 void LinkFrame::WriteUserData(const uint8_t* pSrc, uint8_t* pDest, uint8_t length)
186 {
187  while (length > 0)
188  {
189  uint8_t max = LPDU_DATA_BLOCK_SIZE;
190  uint8_t num = length > max ? max : length;
191  memcpy(pDest, pSrc, num);
192  CRC::AddCrc(pDest, num);
193  pSrc += num;
194  pDest += (num + 2);
