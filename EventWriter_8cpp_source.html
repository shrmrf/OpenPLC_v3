37 bool EventWriter::Write(HeaderWriter& writer, IEventRecorder& recorder, openpal::LinkedListIterator<SOERecord> iterator)
38 {
39  while (iterator.HasNext() && recorder.HasMoreUnwrittenEvents())
40  {
41  auto pCurrent = iterator.Next();
42 
43  if (IsWritable(pCurrent->value))
44  {
45  auto result = LoadHeader(writer, recorder, pCurrent);
46  iterator = result.location;
47 
48  if (result.isFragmentFull)
49  {
50  return false;
51  }
52  }
53  }
54 
55  return true;
56 }

58 EventWriter::Result EventWriter::LoadHeader(HeaderWriter& writer, IEventRecorder& recorder, openpal::ListNode<SOERecord>* pLocation)
59 {
60  switch (pLocation->value.type)
61  {
62  case(EventType::Binary) :
63  return LoadHeaderBinary(writer, recorder, pLocation);
64  case(EventType::DoubleBitBinary) :
65  return LoadHeaderDoubleBinary(writer, recorder, pLocation);
66  case(EventType::Counter):
67  return LoadHeaderCounter(writer, recorder, pLocation);
68  case(EventType::FrozenCounter):
69  return LoadHeaderFrozenCounter(writer, recorder, pLocation);
70  case(EventType::Analog):
71  return LoadHeaderAnalog(writer, recorder, pLocation);
72  case(EventType::BinaryOutputStatus):
73  return LoadHeaderBinaryOutputStatus(writer, recorder, pLocation);
74  case(EventType::AnalogOutputStatus) :
75  return LoadHeaderAnalogOutputStatus(writer, recorder, pLocation);
76  case(EventType::SecurityStat) :
77  return LoadHeaderSecurityStat(writer, recorder, pLocation);
78  default:
79  return Result(false, LinkedListIterator<SOERecord>::Undefined());
80  }
81 }
