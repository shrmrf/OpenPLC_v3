219 template <class T>
220 IINField LoggingHandler::PrintVQT(GroupVariation gv, const ICollection<Indexed<T>>& items)
221 {
222  Indent i(*callbacks);
223  auto logItem = [this, gv](const Indexed<T>& item)
224  {
225  std::ostringstream oss;
226  oss << "[" << item.index << "] - value: " << item.value.value;
227  if (HasFlags(gv))
228  {
229  oss << " flags: 0x" << std::hex << ToHex(item.value.flags.value) << std::dec;
230  }
231  if (HasAbsoluteTime(gv) || HasRelativeTime(gv))
232  {
233  oss << " time: " << ToUTCString(item.value.time);
234  }
235  SIMPLE_LOG_BLOCK(logger, flags::APP_OBJECT_RX, oss.str().c_str());
236  };
237 
238  items.ForeachItem(logItem);
239 
240  return IINField::Empty();
