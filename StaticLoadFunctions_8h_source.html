67 template <class Spec, class IndexType >
68 bool LoadWithRangeIterator(openpal::ArrayView<Cell<Spec>, uint16_t>& view, RangeWriteIterator<IndexType, typename Spec::meas_t>& iterator, Range& range)
69 {
70  const Cell<Spec>& start = view[range.start];
71  uint16_t nextIndex = start.config.vIndex;
72 
73  while (
74  range.IsValid() &&
75  view[range.start].selection.selected &&
76  (view[range.start].selection.variation == start.selection.variation) &&
77  (view[range.start].config.vIndex == nextIndex)
78  )
79  {
80  if (iterator.Write(view[range.start].selection.value))
81  {
82  // deselect the value and advance the range
83  view[range.start].selection.selected = false;
84  range.Advance();
85  ++nextIndex;
86  }
87  else
88  {
89  return false;
90  }
91  }
92 
93  return true;
94 }

127 bool WriteSingleBitfield(openpal::ArrayView<Cell<Spec>, uint16_t>& view, HeaderWriter& writer, Range& range)
128 {
129  auto start = view[range.start].config.vIndex;
130  auto stop = view[range.stop].config.vIndex;
131  auto mapped = Range::From(start, stop);

133  if (mapped.IsOneByte())
