77 ParseResult APDUParser::ParseHeader(RSlice& buffer, openpal::Logger* pLogger, uint32_t count, const ParserSettings& settings, IAPDUHandler* pHandler, IWhiteList* pWhiteList)
78 {
79  ObjectHeader header;
80  auto result = ObjectHeaderParser::ParseObjectHeader(header, buffer, pLogger);
81  if (result != ParseResult::OK)
82  {
83  return result;
84  }
85 
86  const auto GV = GroupVariationRecord::GetRecord(header.group, header.variation);
87 
88  if (GV.enumeration == GroupVariation::UNKNOWN)
89  {
90  FORMAT_LOGGER_BLOCK(pLogger, flags::WARN, "Unknown object %i / %i", GV.group, GV.variation);
91  return ParseResult::UNKNOWN_OBJECT;
92  }
93 
94  // if a white-list is defined and it doesn't validate, exit early
95  if (pWhiteList && !pWhiteList->IsAllowed(count, GV.enumeration, QualifierCodeFromType(header.qualifier)))
96  {
97  FORMAT_LOGGER_BLOCK(
98  pLogger,
99  flags::WARN,
100  "Header (%i) w/ Object (%i,%i) and qualifier (%i) failed whitelist",
101  count,
102  header.group,
103  header.variation,
104  header.qualifier
105  );
106 
107  return ParseResult::NOT_ON_WHITELIST;
108  }
109 
110 
111  return APDUParser::ParseQualifier(buffer, pLogger, HeaderRecord(GV, header.qualifier, count), settings, pHandler);
112 }

114 ParseResult APDUParser::ParseQualifier(RSlice& buffer, openpal::Logger* pLogger, const HeaderRecord& record, const ParserSettings& settings, IAPDUHandler* pHandler)
115 {
116  switch (record.GetQualifierCode())
117  {
118  case(QualifierCode::ALL_OBJECTS) :
119  return HandleAllObjectsHeader(pLogger, record, settings, pHandler);
120 
121  case(QualifierCode::UINT8_CNT) :
122  return CountParser::ParseHeader(buffer, NumParser::OneByte(), settings, record, pLogger, pHandler);
123 
124  case(QualifierCode::UINT16_CNT) :
125  return CountParser::ParseHeader(buffer, NumParser::TwoByte(), settings, record, pLogger, pHandler);
126 
127  case(QualifierCode::UINT8_START_STOP) :
128  return RangeParser::ParseHeader(buffer, NumParser::OneByte(), settings, record, pLogger, pHandler);
129 
130  case(QualifierCode::UINT16_START_STOP) :
131  return RangeParser::ParseHeader(buffer, NumParser::TwoByte(), settings, record, pLogger, pHandler);
132 
133  case(QualifierCode::UINT8_CNT_UINT8_INDEX) :
134  return CountIndexParser::ParseHeader(buffer, NumParser::OneByte(), settings, record, pLogger, pHandler);
135 
136  case(QualifierCode::UINT16_CNT_UINT16_INDEX) :
137  return CountIndexParser::ParseHeader(buffer, NumParser::TwoByte(), settings, record, pLogger, pHandler);
138 
139  case(QualifierCode::UINT16_FREE_FORMAT) :
140  return FreeFormatParser::ParseHeader(buffer, settings, record, pLogger, pHandler);
141 
142  default:
143  FORMAT_LOGGER_BLOCK(pLogger, flags::WARN, "Unknown qualifier %x", record.qualifier);
144  return ParseResult::UNKNOWN_QUALIFIER;
145  }
