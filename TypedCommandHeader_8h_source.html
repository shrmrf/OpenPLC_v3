98 template <class T>
99 bool TypedCommandHeader<T>::Write(HeaderWriter& writer) const
100 {
101  if (m_records.empty())
102  {
103  return false;
104  }
105 
106  auto iter = writer.IterateOverCountWithPrefix<openpal::UInt16, T>(QualifierCode::UINT16_CNT_UINT16_INDEX, m_serializer);
107 
108  for(auto& rec : m_records)
109  {
110  if (!iter.Write(rec.command, rec.index))
111  {
112  return false;
113  }
114  }
115 
116  return iter.IsValid();

119 template <class T>
120 void TypedCommandHeader<T>::ApplySelectResponse(const ICollection<Indexed<T>>& commands)
121 {
122  if (commands.Count() > m_records.size())
123  {
124  return;
125  }
126 
127  uint32_t index = 0;
128 
129  auto visit = [&](const Indexed<T> item) -> void
130  {
131  auto& rec = m_records[index];
132  ++index;
133 
134  if (item.index != rec.index)
135  {
136  return;
137  }
138 
139  if (!item.value.ValuesEqual(rec.command))
140  {
141  rec.state = CommandPointState::SELECT_MISMATCH;
142  return;
143  }
144 
145  if (item.value.status != CommandStatus::SUCCESS)
146  {
147  rec.state = CommandPointState::SELECT_FAIL;
148  rec.status = item.value.status;
149  return;
150  }
151 
152  if (rec.state == CommandPointState::INIT)
153  {
154  rec.state = CommandPointState::SELECT_SUCCESS;
155  }
156  };
157 
158  commands.ForeachItem(visit);
159 }
