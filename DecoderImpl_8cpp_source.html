44 void DecoderImpl::DecodeLPDU(const openpal::RSlice& data)
45 {
46  Indent i(*callbacks);
47 
48  RSlice remaining(data);
49 
50  while (remaining.IsNotEmpty())
51  {
52  auto dest = this->link.WriteBuff();
53 
54  const auto NUM = (remaining.Size() > dest.Size()) ? dest.Size() : remaining.Size();
55 
56  remaining.Take(NUM).CopyTo(dest);
57  link.OnRead(NUM, *this);
58 
59  remaining.Advance(NUM);
60  }

76 void DecoderImpl::DecodeAPDU(const openpal::RSlice& data)
77 {
78  Indent i(*callbacks);
79 
80  FORMAT_HEX_BLOCK(this->logger, flags::APP_HEX_RX, data, 18, 18);
81 
82  if (IsResponse(data))
83  {
84  APDUResponseHeader header;
85  if (APDUHeaderParser::ParseResponse(data, header, &logger))
86  {
87  FORMAT_LOG_BLOCK(this->logger, flags::APP_HEADER_RX,
88  "FIR: %i FIN: %i CON: %i UNS: %i SEQ: %i FUNC: %s IIN: [0x%02x, 0x%02x]",
89  header.control.FIR,
90  header.control.FIN,
91  header.control.CON,
92  header.control.UNS,
93  header.control.SEQ,
94  FunctionCodeToString(header.function),
95  header.IIN.LSB,
96  header.IIN.MSB);
97  if (header.IIN.LSB & 0x01) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.0 - All stations");
98  if (header.IIN.LSB & 0x02) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.1 - Class 1 events");
99  if (header.IIN.LSB & 0x04) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.2 - Class 2 events");
100  if (header.IIN.LSB & 0x08) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.3 - Class 3 events");
101  if (header.IIN.LSB & 0x10) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.4 - Need time");
102  if (header.IIN.LSB & 0x20) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.5 - Local control");
103  if (header.IIN.LSB & 0x40) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.6 - Device trouble");
104  if (header.IIN.LSB & 0x80) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN1.7 - Device restart");
105  if (header.IIN.MSB & 0x01) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.0 - Function code not supported");
106  if (header.IIN.MSB & 0x02) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.1 - Object unknown");
107  if (header.IIN.MSB & 0x04) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.2 - Parameter error");
108  if (header.IIN.MSB & 0x08) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.3 - Event buffer overflow");
109  if (header.IIN.MSB & 0x10) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.4 - Already executing");
110  if (header.IIN.MSB & 0x20) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.5 - Configuration corrupt");
111  if (header.IIN.MSB & 0x40) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.6 - Reserved 1");
112  if (header.IIN.MSB & 0x80) SIMPLE_LOG_BLOCK(this->logger, flags::APP_HEADER_RX, "IIN2.7 - Reserved 2");
113 
114  Indent i(*callbacks);
115  LoggingHandler handler(logger, *callbacks);
116  APDUParser::ParseSinglePass(data.Skip(4), &logger, &handler, nullptr, ParserSettings::Default());
117  }
118  }
119  else
120  {
121  APDUHeader header;
122  if (APDUHeaderParser::ParseRequest(data, header, &logger))
123  {
124 
125  FORMAT_LOG_BLOCK(this->logger, flags::APP_HEADER_RX,
126  "FIR: %i FIN: %i CON: %i UNS: %i SEQ: %i FUNC: %s",
127  header.control.FIR,
128  header.control.FIN,
129  header.control.CON,
130  header.control.UNS,
131  header.control.SEQ,
132  FunctionCodeToString(header.function));
133 
134 
135  Indent i(*callbacks);
136  LoggingHandler handler(logger, *callbacks);
137  auto settings = (header.function == FunctionCode::READ) ? ParserSettings::NoContents() : ParserSettings::Default();
138  APDUParser::ParseSinglePass(data.Skip(2), &logger, &handler, nullptr, settings);
139  }
140  }
