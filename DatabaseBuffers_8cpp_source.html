230 bool DatabaseBuffers::Load(HeaderWriter& writer)
231 {
232  typedef bool (DatabaseBuffers::*LoadFun)(HeaderWriter & writer);
233 
234  const int NUM_TYPE = 8;
235 
236  LoadFun functions[NUM_TYPE] =
237  {
238  &DatabaseBuffers::LoadType<BinarySpec>,
239  &DatabaseBuffers::LoadType<DoubleBitBinarySpec>,
240  &DatabaseBuffers::LoadType<CounterSpec>,
241  &DatabaseBuffers::LoadType<FrozenCounterSpec>,
242  &DatabaseBuffers::LoadType<AnalogSpec>,
243  &DatabaseBuffers::LoadType<BinaryOutputStatusSpec>,
244  &DatabaseBuffers::LoadType<AnalogOutputStatusSpec>,
245  &DatabaseBuffers::LoadType<TimeAndIntervalSpec>
246  };
247 
248  for (int i = 0; i < NUM_TYPE; ++i)
249  {
250  if (!(this->*functions[i])(writer))
251  {
252  // return early because the APDU is full
253  return false;
254  }
255  }
256 
257  return true;
258 }
