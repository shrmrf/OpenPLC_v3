36 void EventBuffer::Unselect()
37 {
38  auto unselect = [this](SOERecord & record)
39  {
40  if (record.selected)
41  {
42  selectedCounts.Decrement(record.clazz, record.type);
43  record.selected = false;
44  }
45 
46  if (record.written)
47  {
48  writtenCounts.Decrement(record.clazz, record.type);
49  record.written = false;
50  }
51 
52  return this->selectedCounts.TotatCount() > 0;
53  };
54 
55  events.While(unselect);
56 }

217 IINField EventBuffer::SelectByClass(const ClassField& field, uint32_t max)
218 {
219  uint32_t num = 0;
220  auto iter = events.Iterate();
221  const uint32_t remaining = totalCounts.NumOfClass(field) - selectedCounts.NumOfClass(field);
222 
223  while (iter.HasNext() && (num < remaining) && (num < max))
224  {
225  auto pNode = iter.Next();
226  if (field.HasEventType(pNode->value.clazz))
227  {
228  pNode->value.SelectDefault();
229  selectedCounts.Increment(pNode->value.clazz, pNode->value.type);
230  ++num;
231  }
232  }
233 
234  return IINField();
235 }

253 {
