78 OutstationState& StateSolicitedConfirmWait::OnConfirm(OContext& ctx, const APDUHeader& header)
79 {
80  if (header.control.UNS)
81  {
82  FORMAT_LOG_BLOCK(ctx.logger, flags::WARN, "received unsolicited confirm while waiting for solicited confirm (seq: %u)", header.control.SEQ);
83  return *this;
84  }
85 
86  if (!ctx.sol.seq.confirmNum.Equals(header.control.SEQ))
87  {
88  FORMAT_LOG_BLOCK(ctx.logger, flags::WARN, "solicited confirm with wrong seq: %u, expected: %u", header.control.SEQ, ctx.sol.seq.confirmNum.Get());
89  return *this;
90  }
91 
92  ctx.history.Reset(); // any time we get a confirm we can treat any request as a new request
93  ctx.confirmTimer.Cancel();
94  ctx.eventBuffer.ClearWritten();
95 
96  if (ctx.rspContext.HasSelection())
97  {
98  return ctx.ContinueMultiFragResponse(AppSeqNum(header.control.SEQ).Next());
99  }
100  else
101  {
102  return StateIdle::Inst();
103  }
104 }

144 OutstationState& StateUnsolicitedConfirmWait::OnConfirm(OContext& ctx, const APDUHeader& header)
145 {
146  if (!header.control.UNS)
147  {
148  FORMAT_LOG_BLOCK(ctx.logger, flags::WARN, "received solicited confirm while waiting for unsolicited confirm (seq: %u)", header.control.SEQ);
149  return *this;
150  }
151 
152  if (!ctx.unsol.seq.confirmNum.Equals(header.control.SEQ))
153  {
154  FORMAT_LOG_BLOCK(ctx.logger, flags::WARN, "unsolicited confirm with wrong seq: %u, expected: %u", header.control.SEQ, ctx.unsol.seq.confirmNum.Get());
155  return *this;
156  }
157 
158  ctx.history.Reset(); // any time we get a confirm we can treat any request as a new request
159  ctx.confirmTimer.Cancel();
160 
161  if (ctx.unsol.completedNull)
162  {
163  ctx.eventBuffer.ClearWritten();
164  }
165  else
166  {
167  ctx.unsol.completedNull = true;
168  }
169 
170  return StateIdle::Inst();
171 }
