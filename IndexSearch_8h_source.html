114 IndexSearch::Result IndexSearch::FindClosestRawIndex(const openpal::ArrayView<Cell<T>, uint16_t>& view, uint16_t vIndex)
115 {
116  if (view.IsEmpty())
117  {
118  return Result(false, 0);
119  }
120  else
121  {
122  uint16_t lower = 0;
123  uint16_t upper = view.Size() - 1;
124 
125  uint16_t midpoint = 0;
126 
127  while (lower <= upper)
128  {
129  midpoint = GetMidpoint(lower, upper);
130 
131  auto index = view[midpoint].config.vIndex;
132 
133  if (index == vIndex)
134  {
135  return Result(true, midpoint);
136  }
137  else
138  {
139  if (index < vIndex) // search the upper array
140  {
141  if (lower < openpal::MaxValue<uint16_t>())
142  {
143  lower = midpoint + 1;
144  }
145  else
146  {
147  //we're at the upper limit
148  return Result(false, midpoint);
149  }
150  }
151  else
152  {
153  if (upper > 0)
154  {
155  upper = midpoint - 1;
156  }
157  else
158  {
159  //we're at the lower limit
160  return Result(false, midpoint);
161  }
162  }
163  }
164  }
165 
166  return Result(false, midpoint);
167  }
168 }
