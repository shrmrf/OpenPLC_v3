60 openpal::ITimer* Executor::Start(const steady_clock_t::time_point& expiration, const openpal::action_t& runnable)
61 {
62  auto timer = std::make_shared<Timer>(this->strand.get_io_service());
63 
64  timer->timer.expires_at(expiration);
65 
66  // neither the executor nor the timer can be deleted while the timer is still active
67  auto callback = [timer, runnable, self = shared_from_this()](const std::error_code & ec)
68  {
69  if (!ec) // an error indicate timer was canceled
70  {
71  runnable();
72  }
73  };
74 
75  timer->timer.async_wait(strand.wrap(callback));
76 
77  return timer.get();
78 }
