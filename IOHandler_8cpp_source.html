137 bool IOHandler::Enable(const std::shared_ptr<opendnp3::ILinkSession>& session)
138 {

142  };
143 
144  const auto iter = std::find_if(sessions.begin(), sessions.end(), matches);
145 
146  if (iter == sessions.end()) return false;

321 void IOHandler::Reset()
322 {
323  if (this->channel)
324  {
325  // shutdown the existing channel and drop the reference to it
326  this->channel->Shutdown();
327  this->channel.reset();
328 
329  ++this->statistics.numClose;
330 
331  this->UpdateListener(ChannelState::CLOSED);
332 
333  // notify any sessions that are online that this layer is offline
334  for (auto& item : this->sessions)
335  {
336  item.LowerLayerDown();
337  }
338  }
339 
340  this->taskLock.SetOffline();
341 
342  // reset the state of the parser
343  this->parser.Reset();
344 
345  // clear any pending tranmissions
346  this->txQueue.clear();
347 }
