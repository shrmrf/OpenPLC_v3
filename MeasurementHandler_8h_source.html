150 IINField MeasurementHandler::ProcessWithCTO(const HeaderRecord& record, const ICollection<Indexed<T>>& values)
151 {
152  if (ctoMode == TimestampMode::INVALID)
153  {
154  FORMAT_LOG_BLOCK(logger, flags::WARN, "No prior CTO objects for %s", GroupVariationToString(record.enumeration));
155  return IINField(IINBit::PARAM_ERROR);
156  }
157 
158  const auto MODE = this->ctoMode;
159  const auto cto = this->commonTimeOccurence;
160 
161  auto transform = [cto](const Indexed<T>& input) -> Indexed<T>
162  {
163  Indexed<T> copy(input);
164  copy.value.time = DNPTime(input.value.time + cto);
165  return copy;
166  };
167 
168  auto adjusted = Map<Indexed<T>, Indexed<T>>(values, transform);
169 
170  return this->LoadValues(record, MODE, adjusted);
171 }
