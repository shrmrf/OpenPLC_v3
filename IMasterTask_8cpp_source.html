62 IMasterTask::ResponseResult IMasterTask::OnResponse(const APDUResponseHeader& response, const openpal::RSlice& objects, openpal::MonotonicTimestamp now)
63 {
64  auto result = ProcessResponse(response, objects);
65 
66  switch (result)
67  {
68  case(ResponseResult::ERROR_BAD_RESPONSE) :
69  this->state = this->OnTaskComplete(TaskCompletion::FAILURE_BAD_RESPONSE, now);
70  this->NotifyResult(TaskCompletion::FAILURE_BAD_RESPONSE);
71  break;
72  case(ResponseResult::ERROR_INTERNAL_FAILURE) :
73  this->state = this->OnTaskComplete(TaskCompletion::FAILURE_INTERNAL_ERROR, now);
74  this->NotifyResult(TaskCompletion::FAILURE_INTERNAL_ERROR);
75  break;
76  case(ResponseResult::OK_FINAL) :
77  this->state = this->OnTaskComplete(TaskCompletion::SUCCESS, now);
78  this->NotifyResult(TaskCompletion::SUCCESS);
79  break;
80  default:
81  break;
82  }
83 
84  return result;
85 }
