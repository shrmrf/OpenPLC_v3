99 template <class Target, class IndexType>
100 IINField CommandResponseHandler::RespondToHeaderWithIterator(QualifierCode qualifier, const DNP3Serializer<Target>& serializer, const ICollection<Indexed<Target>>& values, PrefixedWriteIterator<IndexType, Target>* pIterator)
101 {
102  IINField ret;
103 
104  auto process = [this, pIterator, &ret](const Indexed<Target>& pair)
105  {
106  Target response(pair.value);
107  response.status = this->ProcessCommand(pair.value, pair.index);
108 
109  switch (response.status)
110  {
111  case(CommandStatus::SUCCESS) :
112  ++this->numSuccess;
113  break;
114  case(CommandStatus::NOT_SUPPORTED) :
115  ret.SetBit(IINBit::PARAM_ERROR);
116  break;
117  default:
118  break;
119  }
120 
121  if (pIterator)
122  {
123  pIterator->Write(response, static_cast<typename IndexType::Type>(pair.index));
124  }
125  };
126 
127  values.ForeachItem(process);
128 
129  return ret;
130 }
