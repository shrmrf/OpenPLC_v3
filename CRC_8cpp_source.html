64 uint16_t CRC::CalcCrc(const uint8_t* input, uint32_t length)
65 {
66  uint16_t CRC = 0;
67 
68  for (uint32_t i = 0; i < length; ++i)
69  {
70  uint8_t index = (CRC ^ input[i]) & 0xFF;
71  CRC = crcTable[index] ^ (CRC >> 8);
72  }
73 
74  return ~CRC;
75 }
