141 OutstationState& OContext::OnReceiveSolRequest(const APDUHeader& header, const openpal::RSlice& objects)
142 {
143  // analyze this request to see how it compares to the last request
144  if (this->history.HasLastRequest())
145  {
146  if (this->sol.seq.num.Equals(header.control.SEQ))
147  {
148  if (this->history.FullyEqualsLastRequest(header, objects))
149  {
150  if (header.function == FunctionCode::READ)
151  {
152  return this->state->OnRepeatReadRequest(*this, header, objects);
153  }
154  else
155  {
156  return this->state->OnRepeatNonReadRequest(*this, header, objects);
157  }
158  }
159  else // new operation with same SEQ
160  {
161  return this->ProcessNewRequest(header, objects);
162  }
163  }
164  else // completely new sequence #
165  {
166  return this->ProcessNewRequest(header, objects);
167  }
168  }
169  else
170  {
171  return this->ProcessNewRequest(header, objects);
172  }
173 }

297 void OContext::CheckForUnsolicited()
298 {
299  if (this->CanTransmit() && this->state->IsIdle() && this->params.allowUnsolicited)
300  {
301  if (this->unsol.completedNull)
302  {
303  // are there events to be reported?
304  if (this->params.unsolClassMask.Intersects(this->eventBuffer.UnwrittenClassField()))
305  {
306 
307  auto response = this->unsol.tx.Start();
308  auto writer = response.GetWriter();
309 
310  this->eventBuffer.Unselect();
311  this->eventBuffer.SelectAllByClass(this->params.unsolClassMask);
312  this->eventBuffer.Load(writer);
313 
314  build::NullUnsolicited(response, this->unsol.seq.num, this->GetResponseIIN());
315  this->RestartConfirmTimer();
316  this->state = &StateUnsolicitedConfirmWait::Inst();
317  this->BeginUnsolTx(response.GetControl(), response.ToRSlice());
318  }
319  }
320  else
321  {
322  // send a NULL unsolcited message
323  auto response = this->unsol.tx.Start();
324  build::NullUnsolicited(response, this->unsol.seq.num, this->GetResponseIIN());
325  this->RestartConfirmTimer();
326  this->state = &StateUnsolicitedConfirmWait::Inst();
327  this->BeginUnsolTx(response.GetControl(), response.ToRSlice());
328  }
329  }

344 {

356 OutstationState& OContext::RespondToReadRequest(const APDUHeader& header, const openpal::RSlice& objects)
357 {
358  this->history.RecordLastProcessedRequest(header, objects);
359 
360  auto response = this->sol.tx.Start();
361  auto writer = response.GetWriter();
362  response.SetFunction(FunctionCode::RESPONSE);
363  auto result = this->HandleRead(objects, writer);
364  result.second.SEQ = header.control.SEQ;
365  this->sol.seq.confirmNum = header.control.SEQ;
366  response.SetControl(result.second);
367  response.SetIIN(result.first | this->GetResponseIIN());
368  this->BeginResponseTx(response.GetControl(), response.ToRSlice());
369 
370  if (result.second.CON)
371  {
372  this->RestartConfirmTimer();
373  return StateSolicitedConfirmWait::Inst();
374  }
375  else
376  {
377  return StateIdle::Inst();
378  }

427 void OContext::ParseHeader(const openpal::RSlice& apdu)
428 {
429  FORMAT_HEX_BLOCK(this->logger, flags::APP_HEX_RX, apdu, 18, 18);
430 
431  APDUHeader header;
432  if (!APDUHeaderParser::ParseRequest(apdu, header, &this->logger))
433  {
434  return;
435  }
436 
437  FORMAT_LOG_BLOCK(this->logger, flags::APP_HEADER_RX,
438  "FIR: %i FIN: %i CON: %i UNS: %i SEQ: %i FUNC: %s",
439  header.control.FIR,
440  header.control.FIN,
441  header.control.CON,
442  header.control.UNS,
443  header.control.SEQ,
444  FunctionCodeToString(header.function));
445 
446  // outstations should only process single fragment messages that don't request confirmation
447  if (!header.control.IsFirAndFin() || header.control.CON)
448  {
449  SIMPLE_LOG_BLOCK(this->logger, flags::WARN, "Ignoring fragment. Request must be FIR/FIN/!CON");
450  return;
451  }
452 
453  auto objects = apdu.Skip(APDU_REQUEST_HEADER_SIZE);
454 
455  this->ProcessAPDU(apdu, header, objects);
456 }

495 IINField OContext::HandleNonReadResponse(const APDUHeader& header, const openpal::RSlice& objects, HeaderWriter& writer)
496 {
497  switch (header.function)
498  {
499  case(FunctionCode::WRITE) :
500  return this->HandleWrite(objects);
501  case(FunctionCode::SELECT) :
502  return this->HandleSelect(objects, writer);
503  case(FunctionCode::OPERATE) :
504  return this->HandleOperate(objects, writer);
505  case(FunctionCode::DIRECT_OPERATE) :
506  return this->HandleDirectOperate(objects, OperateType::DirectOperate, &writer);
507  case(FunctionCode::COLD_RESTART) :
508  return this->HandleRestart(objects, false, &writer);
509  case(FunctionCode::WARM_RESTART) :
510  return this->HandleRestart(objects, true, &writer);
511  case(FunctionCode::ASSIGN_CLASS) :
512  return this->HandleAssignClass(objects);
513  case(FunctionCode::DELAY_MEASURE) :
514  return this->HandleDelayMeasure(objects, writer);
515  case(FunctionCode::DISABLE_UNSOLICITED) :
516  return this->params.allowUnsolicited ? this->HandleDisableUnsolicited(objects, writer) : IINField(IINBit::FUNC_NOT_SUPPORTED);
517  case(FunctionCode::ENABLE_UNSOLICITED) :
518  return this->params.allowUnsolicited ? this->HandleEnableUnsolicited(objects, writer) : IINField(IINBit::FUNC_NOT_SUPPORTED);
519  default:
520  return IINField(IINBit::FUNC_NOT_SUPPORTED);
521  }

524 Pair<IINField, AppControlField> OContext::HandleRead(const openpal::RSlice& objects, HeaderWriter& writer)
525 {
526  this->rspContext.Reset();
527  this->eventBuffer.Unselect(); // always un-select any previously selected points when we start a new read request
528  this->database.GetStaticSelector().Unselect();
529 
530  ReadHandler handler(this->database.GetStaticSelector(), this->eventBuffer);
531  auto result = APDUParser::Parse(objects, handler, &this->logger, ParserSettings::NoContents()); // don't expect range/count context on a READ
532  if (result == ParseResult::OK)
533  {
534  auto control = this->rspContext.LoadResponse(writer);
535  return Pair<IINField, AppControlField>(handler.Errors(), control);
536  }
537  else
538  {

540  return Pair<IINField, AppControlField>(IINFromParseResult(result), AppControlField(true, true, false, false));

641 IINField OContext::HandleRestart(const openpal::RSlice& objects, bool isWarmRestart, HeaderWriter* pWriter)
642 {
643  if (objects.IsEmpty())
644  {
645  auto mode = isWarmRestart ? this->application->WarmRestartSupport() : this->application->ColdRestartSupport();
646 
647  switch (mode)
648  {
649  case(RestartMode::UNSUPPORTED) :
650  return IINField(IINBit::FUNC_NOT_SUPPORTED);
651  case(RestartMode::SUPPORTED_DELAY_COARSE) :
652  {

658  pWriter->WriteSingleValue<UInt8>(QualifierCode::UINT8_CNT, coarse);
659  }
660  return IINField::Empty();

662  default:

664  auto delay = isWarmRestart ? this->application->WarmRestart() : this->application->ColdRestart();
665  if (pWriter)
666  {
667  Group52Var2 fine;
668  fine.time = delay;
669  pWriter->WriteSingleValue<UInt8>(QualifierCode::UINT8_CNT, fine);
670  }
671  return IINField::Empty();
672  }
673  }
674  }
675  else
676  {
677  // there shouldn't be any trailing headers in restart requests, no need to even parse
678  return IINField(IINBit::PARAM_ERROR);
