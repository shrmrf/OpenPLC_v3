45 bool SerialTimeSyncTask::BuildRequest(APDURequest& request, uint8_t seq)
46 {
47  if (delay < 0)
48  {
49  start = pApplication->Now();
50  build::MeasureDelay(request, seq);
51  }
52  else
53  {
54  auto now = pApplication->Now();
55  Group50Var1 time;
56  time.time = DNPTime(now.msSinceEpoch + delay);
57  request.SetFunction(FunctionCode::WRITE);
58  request.SetControl(AppControlField::Request(seq));
59  auto writer = request.GetWriter();
60  writer.WriteSingleValue<UInt8, Group50Var1>(QualifierCode::UINT8_CNT, time);
61  }
62 
63  return true;
64 }

82 IMasterTask::ResponseResult SerialTimeSyncTask::OnResponseDelayMeas(const APDUResponseHeader& response, const openpal::RSlice& objects)
83 {
84  if (ValidateSingleResponse(response))
85  {
86  TimeSyncHandler handler;
87  auto result = APDUParser::Parse(objects, handler, &logger);
88  if (result == ParseResult::OK)
89  {
90  uint16_t rtuTurnAroundTime;
91  if (handler.GetTimeDelay(rtuTurnAroundTime))
92  {
93  auto now = pApplication->Now();
94  auto sendReceieveTime = now.msSinceEpoch - start.msSinceEpoch;
95 
96  // The later shouldn't happen, but could cause a negative delay which would result in a weird time setting
97  delay = (sendReceieveTime >= rtuTurnAroundTime) ? (sendReceieveTime - rtuTurnAroundTime) / 2 : 0;
98 
99  return ResponseResult::OK_REPEAT;
100  }
101  else
102  {
103  return ResponseResult::ERROR_BAD_RESPONSE;
104  }
105  }
