37 /*
38  The sequence of events list is a doubly linked-list implemented
39  in a finite array. The list is desired for O(1) remove operations from
40  arbitrary parts of the list depending on what the user asks for in terms
41  of event type or Class1/2/3.
42 
43  At worst, selection is O(n) in the SOE length but it has some type/class
44  tracking to avoid looping over the SOE list when there are no more events matching
45  the selection criteria.
46 */

169 template <class Spec>
170 void EventBuffer::UpdateAny(const Event<Spec>& evt)
171 {
172  auto maxForType = config.GetMaxEventsForType(Spec::EventTypeEnum);
173 
174  if (maxForType > 0)
175  {
176  auto currentCount = totalCounts.NumOfType(Spec::EventTypeEnum);
177 
178  if (currentCount >= maxForType || events.IsFull())
179  {
180  this->overflow = true;
181  RemoveOldestEventOfType(Spec::EventTypeEnum);
182  }
183 
184  // Add the event, the Reset() ensures that selected/written == false
185  events.Add(SOERecord(evt.value, evt.index, evt.clazz, evt.variation))->value.Reset();
186  totalCounts.Increment(evt.clazz, Spec::EventTypeEnum);
187  }
188 }

190 template <class Spec>
191 uint32_t EventBuffer::GenericSelectByType(uint32_t max, bool useDefault, typename Spec::event_variation_t var)
192 {
193  uint32_t num = 0;
194  auto iter = events.Iterate();
195  const uint32_t remaining = totalCounts.NumOfType(Spec::EventTypeEnum) - selectedCounts.NumOfType(Spec::EventTypeEnum);
196 
197  while (iter.HasNext() && (num < remaining) && (num < max))
198  {
199  auto pNode = iter.Next();
200 
201  if (pNode->value.type == Spec::EventTypeEnum)
202  {
203  if (useDefault)
204  {
205  pNode->value.SelectDefault();
206  }
207  else
208  {
209  pNode->value.Select(var);
210  }
211 
212  selectedCounts.Increment(pNode->value.clazz, pNode->value.type);
213  ++num;
214  }
215  }
216 
217  return num;
218 }
